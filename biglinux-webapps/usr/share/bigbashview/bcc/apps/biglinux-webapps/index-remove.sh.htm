#!/bin/bash #Translation export TEXTDOMAINDIR="/usr/share/locale" export TEXTDOMAIN=biglinux-webapps echo ' BigLinux WebApps

' $"Remover WebApps BigLinux:" '



' DESKTOP="$(ls $HOME/.local/share/applications/*.desktop | grep "webapp-biglinux-custom" | sort )" if [ "$DESKTOP" != "" ]; then echo '

' for i in $DESKTOP; do if [ "$(grep -E "(Exec=firefox|Exec=waterfox-latest)" $i)" != "" ];then DESKEXEC="$(grep "Exec=" $i | sed 's|Exec=||')" DESKURL="$(grep "\-new-instance" $DESKEXEC | cut -d' ' -f9 | sed 's|^"||;s|"$||')" elif [ "$(grep "Exec=falkon" $i)" != "" ];then DESKURL="$(grep "Exec=" $i | cut -d' ' -f4)" elif [ "$(grep "Exec=min" $i)" != "" ];then DESKURL="$(grep "Exec=" $i | cut -d' ' -f2)" else DESKURL="$(grep "Exec=" $i | awk -F'--app=' '{print $NF}')" fi DESKNAME="$(grep "Name=" $i | sed 's|Name=||')" DESKICON="$(grep "Icon=" $i | sed 's|Icon=||')" if [ "$DESKICON" = "internet-web-browser" ];then DESKICON="default.png" fi echo '

' done echo '

'$"Voltar"'

' else echo $"Nenhum WebApp BigLinux instalado por você!" '

' echo $"Se você deseja instalar um novo
WebApp BigLinux, clique no botão Instalar:" '


' echo ''$"Voltar"' '$"Instalar"' ' fi
